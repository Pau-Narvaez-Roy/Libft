/* TEST: ft_lstmap.c */
#include "libft.h"
#include <stdio.h>
#include <stdlib.h>

void *duplicate(void *content)
{
	char *str = (char *)content;
	char *dup = malloc(10);
	int i = 0;
	while (str[i])
	{
		dup[i] = str[i];
		i++;
	}
	dup[i] = '\0';
	return (dup);
}

void del(void *content)
{
	free(content);
}

int main(void)
{
	t_list *head = ft_lstnew("a");
	ft_lstadd_back(&head, ft_lstnew("b"));
	t_list *new_list = ft_lstmap(head, duplicate, del);
	printf("ft_lstmap tests:\n");
	printf("New list first: %s\n", (char *)new_list->content);
	return (0);
}
